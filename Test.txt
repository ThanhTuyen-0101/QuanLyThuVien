        [TestMethod]
        public void KichBan3_KiemTra_TinhPhiPhat_MuonTraSach()
        {
            using (var app = Application.Launch(appPath))
            {
                using (var automation = new UIA3Automation())
                {
                    var window = app.GetMainWindow(automation);

                    // Đăng nhập
                    window.FindFirstDescendant(cf => cf.ByAutomationId("txttendangnhap"))?.AsTextBox().Enter("Tuyen");
                    window.FindFirstDescendant(cf => cf.ByAutomationId("txtmatkhaudangnhap"))?.AsTextBox().Enter("1234@Ntt");
                    window.FindFirstDescendant(cf => cf.ByAutomationId("btndagnhap"))?.AsButton().Click();
                    System.Threading.Thread.Sleep(1500);
                    Keyboard.Press(VirtualKeyShort.ENTER);
                    System.Threading.Thread.Sleep(1500);

                    // Vào form Mượn Trả
                    var trangChu = app.GetMainWindow(automation);
                    var btnMuonTra = trangChu.FindFirstDescendant(cf => cf.ByAutomationId("btnmuontra"))?.AsButton();
                    btnMuonTra?.Click();
                    System.Threading.Thread.Sleep(1500);

                    var formMuonTra = app.GetMainWindow(automation);
                    var txtNgayMuon = formMuonTra.FindFirstDescendant(cf => cf.ByAutomationId("txtDateMuon"))?.AsTextBox();
                    var chkTraSach = formMuonTra.FindFirstDescendant(cf => cf.ByAutomationId("Checktra"))?.AsCheckBox();
                    var txtTienPhat = formMuonTra.FindFirstDescendant(cf => cf.ByAutomationId("txtTienPhat"))?.AsTextBox();

                    Assert.IsNotNull(chkTraSach, "Không tìm thấy Checkbox Trả sách");
                    Assert.IsNotNull(txtNgayMuon, "Không tìm thấy txtDateMuon");
                    Assert.IsNotNull(txtTienPhat, "Không tìm thấy txtTienPhat");

                    // 🔥 FIX 1: CLICK VÀO TEXTBOX TRƯỚC KHI SET DATE
                    txtNgayMuon.Click();
                    string ngayMuonQuaHan = System.DateTime.Now.AddDays(-30).ToString("dd/MM/yyyy");
                    txtNgayMuon.Enter(ngayMuonQuaHan); // Enter thay vì Text=

                    // 🔥 FIX 2: TAB RA ĐỂ TRIGGER LOSTFOCUS + CALCULATION
                    Keyboard.Press(VirtualKeyShort.TAB);
                    System.Threading.Thread.Sleep(1000);

                    // Check Trả sách
                    chkTraSach.Click();
                    System.Threading.Thread.Sleep(2000); // Chờ tính phí

                    // 🔥 FIX 3: TRIGGER TÍNH TOÁN THÊM (nếu cần)
                    Keyboard.Press(VirtualKeyShort.TAB);
                    System.Threading.Thread.Sleep(500);

                    // Đọc kết quả
                    string tienPhatHienTai = txtTienPhat.Text.Trim();
                    Console.WriteLine($"Ngày mượn: {ngayMuonQuaHan}");
                    Console.WriteLine($"Tiền phạt: '{tienPhatHienTai}'");

                    // Assert linh hoạt hơn
                    Assert.IsTrue(
                        !string.IsNullOrEmpty(tienPhatHienTai) &&
                        tienPhatHienTai != "0" &&
                        double.TryParse(tienPhatHienTai, out double phi) && phi > 0,
                        $"Lỗi logic: Ngày quá hạn 30 ngày nhưng tiền phạt vẫn '{tienPhatHienTai}'!");

                    Console.WriteLine("PASS: Tính phí phạt thành công!");
                    app.Close();
                }
            }
        }
